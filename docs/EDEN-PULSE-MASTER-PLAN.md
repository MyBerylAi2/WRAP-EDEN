### Microsoft Agent Framework Implementation

```python
# Eden Pulse runs as a WORKFLOW with 5 AGENTS
from agent_framework import Agent, Workflow, Edge, ModelClient
from agent_framework.models import HuggingFaceClient

# Each Heartbeat is an Agent with its own model + tools
archivist = Agent(
    name="Dr. Amara Okafor",
    instructions="You are THE ARCHIVIST...",
    model=HuggingFaceClient(model="Qwen/Qwen2.5-3B-Instruct"),
    tools=[arxiv_fetch, hf_model_card, pdf_extract, github_readme]
)

# The workflow connects them sequentially
workflow = Workflow()
workflow.add_agent("archivist", archivist)
workflow.add_agent("analyst", analyst)
workflow.add_agent("prophet", prophet)
workflow.add_agent("synthesist", synthesist)
workflow.add_agent("journalist", journalist)

# Sequential flow: each agent builds on the previous
workflow.add_edge(Edge("archivist", "analyst"))
workflow.add_edge(Edge("analyst", "prophet"))
workflow.add_edge(Edge("prophet", "synthesist"))
workflow.add_edge(Edge("synthesist", "journalist"))
```
